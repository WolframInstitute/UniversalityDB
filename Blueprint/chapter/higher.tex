\chapter{Higher Structures}\label{ch:higher}

\begin{definition}[Simulation conjugacy]\label{sim-conjugacy}
\notready
\uses{simulation-def}
When two distinct simulation encodings $e_1, e_2 : A \to B$ exist, they
give rise to 2-morphisms (natural transformations between the encodings).
$\UG$ extends to a 2-category:
\begin{itemize}
\item Objects: computational systems
\item 1-morphisms: simulation encodings
\item 2-morphisms: encoding conjugacies, overhead domination, proof homotopies
\end{itemize}
This connects to HoTT~\cite{hottbook}: paths in the space of proofs.
\end{definition}
