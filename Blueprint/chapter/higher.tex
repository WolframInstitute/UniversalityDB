\chapter{Higher Structures}\label{ch:higher}

\begin{definition}[Simulation conjugacy]\label{sim-conjugacy}
\lean{CompSystem.SimConjugacy}
\leanok
\uses{simulation-def}
Distinct proofs between the same pair give rise to 2-morphisms. $\UG$ extends
to a 2-category: objects = systems, 1-morphisms = simulations, 2-morphisms =
transformations (encoding conjugacy, overhead domination, proof homotopy).
Connects to HoTT~\cite{hottbook}.
\end{definition}
