\chapter{Open Questions}\label{ch:open}

\begin{question}\label{q-gs-tm-sorry}
\uses{prop:gs-tm}
Close the remaining \texttt{sorry} in GS~$\to$~TM (Moore Theorem~8). The step
simulation proof requires three phase lemmas (read, write, shift) composed
over the \texttt{buildTransition} function.
\end{question}

\begin{question}\label{q-cocke-minsky}
\uses{prop:cocke-minsky}
Prove \texttt{CockeMinskyStepSimulation}: one TM step maps to a bounded number
of tag steps under the Cocke--Minsky encoding.
\end{question}
