\chapter{Computational Systems}\label{ch:systems}

\begin{definition}[Computational machine]\label{comp-system}
\lean{ComputationalMachine}
\leanok
A computational machine $M$ consists of a configuration type $\mathrm{Config}_M$
and a partial step function $\mathrm{step}_M : \mathrm{Config}_M \to
\mathrm{Config}_M \cup \{\bot\}$.
A configuration $c$ \emph{halts} if $\mathrm{step}_M^n(c) = \bot$ for some $n$.
\end{definition}

\section{Turing Machines}

A Turing machine $\mathrm{TM}(s,k)$ has $s$ states and $k$ symbols.
The transition function maps (state, symbol) to (new state, write symbol,
head direction). State~0 is the halt state.

\begin{definition}[One-sided TM]\label{one-sided-tm}
\lean{TuringMachine.Machine}
\lean{TuringMachine.decodeRule}
\leanok
\uses{comp-system}
A TM parameterized by \texttt{numberOfStates} and \texttt{numberOfSymbols}.
\texttt{decodeRule} converts a Wolfram-style rule number to explicit transitions,
enabling systematic enumeration of all $(s,k)$ machines.
\end{definition}

\begin{definition}[Bi-infinite TM]\label{bi-tm}
\lean{BiInfiniteTuringMachine.Configuration}
\lean{BiInfiniteTuringMachine.step}
\leanok
\uses{comp-system}
Configuration = (state, left tape, head symbol, right tape) on a bi-infinite
tape. Includes \texttt{evaluate} (fuel-bounded), \texttt{exactSteps}, and
\texttt{Halts}. Used for simulation proofs.
\end{definition}

\begin{example}[Wolfram's $(2,3)$ TM]\label{wolfram-23-tm}
\lean{BiInfiniteTuringMachine.wolfram23}
\leanok
\uses{bi-tm}
The smallest known universal Turing machine: 2 states, 3 symbols, 6 transition
rules. Verified to run 10 and 20 steps from the standard initial configuration
(\texttt{wolfram23Runs10}, \texttt{wolfram23Runs20}).
Smith~\cite{smith2007universality} proved its universality.
\end{example}

\section{Elementary Cellular Automata}

An elementary cellular automaton (ECA) has 2 states and nearest-neighbor rule on
a periodic tape, parameterized by a rule number $r \in \{0,\ldots,255\}$.

\begin{definition}[ECA]\label{eca-def}
\lean{ruleTable, step}
\leanok
\uses{comp-system}
\texttt{ruleTable} converts a rule number and 3-cell neighborhood to an output
bit. \texttt{step} applies the rule to all cells in parallel on a periodic tape
of size $n$. Built-in rules: \texttt{rule110}, \texttt{rule124}, \texttt{rule137}.
\end{definition}

\section{Tag Systems}

\begin{definition}[2-tag system]\label{tag-def}
\lean{Tag, Tag.step, Tag.Halts}
\leanok
\uses{comp-system}
A 2-tag system with alphabet of size $k$ and production function
$P : \mathrm{Fin}(k) \to \mathrm{Fin}(k)^*$. At each step:
read the first symbol $a$, append $P(a)$, delete the first 2 symbols.
Halts when the word length drops below~2.
\end{definition}

\begin{definition}[Cyclic tag system]\label{cts-def}
\lean{CyclicTagSystem}
\lean{CyclicTagSystem.step}
\lean{CyclicTagSystem.Halts}
\leanok
\uses{comp-system}
Binary data with a cyclic list of appendants. At each step: remove the first
bit; if it was~1, append the current appendant; advance the phase (mod number
of appendants). Halts when the data is empty.
\end{definition}

\section{Generalized Shifts}

\begin{definition}[Generalized shift]\label{gs-def}
\lean{GeneralizedShift.Machine}
\lean{GeneralizedShift.step}
\leanok
\uses{comp-system}
Moore's generalized shift~\cite{moore1991generalized}: a stateless machine on a
bi-infinite tape with alphabet size $n$ and window width $w$. At each step:
read the $w$-cell window; if active, replace the window contents, shift the
window by some magnitude in a given direction. Inactive windows halt.
\end{definition}

\section{Other Formalisms}

Not yet formalized:
\begin{itemize}
\item \textbf{Register machines}: finite-state controller on unbounded integer
  registers. Simulate TMs with $\oh{1}$ overhead.
\item \textbf{Planar billiards}: Miranda--Ramos~\cite{miranda2025billiards}:
  billiard trajectories simulate TMs with $\tovh = \oh{|Q|}$.
\item \textbf{Game of Life}: Rendell~\cite{rendell2002turing}: explicit TM
  construction with $\sovh \approx 10^3$.
\end{itemize}
