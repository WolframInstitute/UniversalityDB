\section{ECA Conjugation Simulations}

The 256 ECAs partition into equivalence classes under mirror ($M$), complement ($C$),
and $MC$. Each gives a simulation with $\sovh=1$, $\tovh=1$.

\begin{proposition}[Mirror simulation]\label{mirror-sim}
\lean{ECA.mirror_property, ECA.mirror_sim_steps}
\leanok
\uses{simulation-def}
$\mathrm{step}_r(\mathrm{rev}(c)) = \mathrm{rev}(\mathrm{step}_{\bar{r}}(c))$
for any periodic tape $c$.
\end{proposition}

Rule 110's class: $\{110, 124, 137, 193\}$.
Lean: pointwise by \texttt{fin\_cases} + \texttt{decide};
full theorem by \texttt{native\_decide} for small sizes.

\section{TM $\leftrightarrow$ Generalized Shift (Moore 1991)}

\begin{proposition}[TM $\to$ GS, Moore Theorem 7]\label{prop:tm-gs}
\lean{TuringMachineToGeneralizedShift.fromBiTM, TuringMachineToGeneralizedShift.stepCommutes, TuringMachineToGeneralizedShift.decodeEncode, TuringMachineToGeneralizedShift.mooreHaltingForward}
\leanok
\uses{simulation-def, overhead-tuple}
Any $(s,k)$ TM is simulated by a GS with window 3, alphabet $sk+k$,
$\sovh = 1$, $\tovh = 1$.
The encoding merges state into the cell under the head:
symbol $k + (q-1)k + c$ = state $q$ reading color $c$.
Fully verified in Lean with zero \texttt{sorry}.
\end{proposition}

\begin{proposition}[GS $\to$ TM, Moore Theorem 8]\label{prop:gs-tm}
\lean{GeneralizedShiftToTuringMachine.toBiTM, GeneralizedShiftToTuringMachine.buildTransition}
\notready
\uses{simulation-def, overhead-tuple}
Any GS with window $w$ and max shift $m$ is simulated by a TM with
$\sovh = 1$, $\tovh \leq 2(w-1) + m$.
The TM uses read/write/shift phases to simulate one GS step.
Construction defined in Lean; step simulation proof pending.
\end{proposition}
